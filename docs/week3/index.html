x <- elephants$Fore_Feet_Circumference
y <- elephants$Height
ar <- (max(x)-min(x))/(max(y)-min(y))
ggplot(data=elephants,
aes(x=Fore_Feet_Circumference,
y=Height)) + geom_point() + coord_fixed(ratio=ar)
